# Define a function to categorize resolution descriptions into broader groups
def categorize_resolution(description):
    if "action" in description.lower() and "no" not in description.lower():
        return "Action Taken"
    elif "no evidence" in description.lower() or "not necessary" in description.lower():
        return "No Action Necessary"
    elif "unable to gain entry" in description.lower():
        return "Unable to Gain Entry"
    elif "summons" in description.lower():
        return "Summons Issued"
    elif "duplicate" in description.lower():
        return "Duplicate Complaint"
    elif "insufficient" in description.lower():
        return "Insufficient Information"
    elif "request" in description.lower() or "withdrawn" in description.lower():
        return "Complaint Withdrawn"
    else:
        return "Other"

# Apply categorization
data['Resolution.Category'] = data['Resolution.Description'].apply(categorize_resolution)

# Group by borough and resolution category, and count occurrences
resolution_outcomes_by_borough = data.groupby(['Borough', 'Resolution.Category']).size().unstack(fill_value=0)

# Plot the data
fig, ax = plt.subplots(figsize=(14, 8))
resolution_outcomes_by_borough.plot(kind='bar', stacked=True, ax=ax)
ax.set_title('Complaint Resolution Outcomes by Borough', fontsize=16)
ax.set_xlabel('Borough', fontsize=14)
ax.set_ylabel('Number of Complaints', fontsize=14)
ax.legend(title='Resolution Category')
plt.xticks(rotation=45)
plt.show()
